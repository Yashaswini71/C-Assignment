1.Write a program to print the below pattern using while loop:

sol:#include<stdio.h>
void main()
{
    int i,j,k,space,rows;
    printf("enter no of rows:");
    scanf("%d",&rows);
        
    for(i=0;i<=rows;i++)
    {
        if(i==0)
        {
            for(k=0;k<=rows;k++)
            {
                printf(" ");
            }
        }
        for(space=0;space<=rows-i;space++)
        {
            printf(" ");
        }
        for(j=1;j<(2*i-1);j++)
        {
            if(j==1)
                printf("/");
            else if(j==(2*i-2))
                printf("%c",'\\');
            else
                printf(" ");
        }
        printf("\n");
    }
    for(i=0;i<rows;i++)
    {
        printf(" _");
    }
}

4.A sequence must be constraint on its values.

sol:
#include<stdio.h>
void main()
{
    int sequence[]={1,2,2,3,3,3,4,4,4,4},input[10],i,validity;
    for(i=0;i<10;i++)
    {
        printf("\nenter the number");scanf("%d",&input[i]);
        (input[i]==sequence[i])?(validity=1):(validity=0);
    }
    printf("\n\ncorrect sequence:\t");
    for(i=0;i<10;i++)
        printf("%d",sequence[i]);
    printf("\ninput sequence:\t");
    for(i=0;i<10;i++)
        printf("%d",input[i]);
    (validity==1)?printf("\nseqence is valid"):printf("\nseqence is invalid");
}

5.Write a c program to print all even and odd numbers btwn 1-100  
sol:
#include<stdio.h>
void main()
{
    printf("Even numbers are:");
    for(int i=1;i<101;i++)
    {
        if((i%2)==0)
            printf("%d ",i);
    }
    printf("\nOdd numbers are:");
    for(int i=1;i<101;i++)
    {
        if((i%2)!=0)
            printf("%d ",i);
    }
}
